## Bloque C · Pull-to-refresh (1 día) — paralelizable

### Tarea C.1 · Hook reutilizable

- [ ] **Paso 1:** Crear `src/frontend/src/hooks/usePullToRefresh.ts`.

```ts
// src/frontend/src/hooks/usePullToRefresh.ts
import { useEffect, useRef, useState } from 'react';
import * as haptics from '../platform/haptics';

interface Options {
  onRefresh: () => Promise<void> | void;
  threshold?: number;       // px de pull antes de trigger
  disabled?: boolean;
}

export function usePullToRefresh({ onRefresh, threshold = 80, disabled = false }: Options) {
  const startY = useRef(0);
  const [pull, setPull] = useState(0);
  const [refreshing, setRefreshing] = useState(false);
  const containerRef = useRef<HTMLDivElement | null>(null);

  useEffect(() => {
    if (disabled || !containerRef.current) return;
    const el = containerRef.current;

    const onTouchStart = (e: TouchEvent) => {
      if (el.scrollTop > 0) return;
      startY.current = e.touches[0].clientY;
    };

    const onTouchMove = (e: TouchEvent) => {
      if (refreshing || el.scrollTop > 0) return;
      const dy = e.touches[0].clientY - startY.current;
      if (dy > 0) {
        setPull(Math.min(dy * 0.5, threshold * 1.5));  // resistencia
        if (dy >= threshold && !refreshing) {
          haptics.trigger('light');
        }
      }
    };

    const onTouchEnd = async () => {
      if (pull >= threshold && !refreshing) {
        setRefreshing(true);
        haptics.trigger('medium');
        try { await onRefresh(); } finally {
          setRefreshing(false);
          setPull(0);
        }
      } else {
        setPull(0);
      }
    };

    el.addEventListener('touchstart', onTouchStart, { passive: true });
    el.addEventListener('touchmove', onTouchMove, { passive: true });
    el.addEventListener('touchend', onTouchEnd, { passive: true });
    return () => {
      el.removeEventListener('touchstart', onTouchStart);
      el.removeEventListener('touchmove', onTouchMove);
      el.removeEventListener('touchend', onTouchEnd);
    };
  }, [disabled, onRefresh, pull, refreshing, threshold]);

  return { containerRef, pull, refreshing };
}
```

```tsx
import { RefreshCw } from 'lucide-react';

interface Props { pull: number; refreshing: boolean; threshold?: number; }

export function PullToRefreshIndicator({ pull, refreshing, threshold = 80 }: Props) {
  if (pull === 0 && !refreshing) return null;
  const progress = Math.min(pull / threshold, 1);
  return (
    <div
      aria-hidden
      style={{ height: pull, transition: refreshing ? 'none' : 'height 200ms' }}
      className="flex items-center justify-center bg-surface-card overflow-hidden"
    >
      <RefreshCw
        className={`w-5 h-5 text-brand-purple ${refreshing ? 'animate-spin' : ''}`}
        style={{ transform: `rotate(${progress * 180}deg)` }}
      />
    </div>
  );
}
```

### Tarea E.2 · Optimistic UI en mutations frecuentes

Para acciones que el usuario hace > 5 veces/sesión:

1. **Mark task done** (Tasks.tsx): aplicar `setQueryData` optimista antes de await mutation.
2. **React to journal entry**: misma cosa.
3. **Accept negotiation**: optimista — sigue mostrando "Procesando..." si tarda > 500 ms.

- [ ] **Paso 1:** Patrón Tanstack Query `onMutate` + `onError` rollback.

```ts
const completeTask = useMutation({
  mutationFn: api.tasks.complete,
  onMutate: async (taskId) => {
    await queryClient.cancelQueries({ queryKey: ['tasks'] });
    const previous = queryClient.getQueryData(['tasks']);
    queryClient.setQueryData(['tasks'], (old: any) =>
      old?.map((t: any) => t.id === taskId ? { ...t, status: 'completed' } : t)
    );
    return { previous };
  },
  onError: (err, _, ctx) => {
    queryClient.setQueryData(['tasks'], ctx?.previous);
    haptics.trigger('error');
  },
  onSettled: () => queryClient.invalidateQueries({ queryKey: ['tasks'] })
});
```

```ts
import { useEffect } from 'react';
import { useNavigate } from 'react-router-dom';
import * as install from '../platform/install';

export function useSwipeBack(enabled = true) {
  const navigate = useNavigate();

  useEffect(() => {
    if (!enabled || !install.isIOSDevice()) return;

    let startX = 0, startY = 0, startT = 0;
    const onStart = (e: TouchEvent) => {
      if (e.touches[0].clientX > 20) return;  // solo desde borde izquierdo
      startX = e.touches[0].clientX;
      startY = e.touches[0].clientY;
      startT = Date.now();
    };
    const onEnd = (e: TouchEvent) => {
      if (startX === 0) return;
      const dx = e.changedTouches[0].clientX - startX;
      const dy = Math.abs(e.changedTouches[0].clientY - startY);
      const dt = Date.now() - startT;
      if (dx > 80 && dy < 40 && dt < 500) {
        navigate(-1);
      }
      startX = 0;
    };

    document.addEventListener('touchstart', onStart, { passive: true });
    document.addEventListener('touchend', onEnd, { passive: true });
    return () => {
      document.removeEventListener('touchstart', onStart);
      document.removeEventListener('touchend', onEnd);
    };
  }, [enabled, navigate]);
}
```
