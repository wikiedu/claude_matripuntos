# v1.6 · La Personalidad — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Convertir Matripuntos de "herramienta de gestión" a "espacio compartido con personalidad" añadiendo frase del día con cascada emocional, mood ligero por persona con caducidad 24h, y avatar consolidado, todo con cobertura de tests + QA dos cuentas + cierre de deuda de tests backend de v1.5.

**Architecture:** Backend introduce tabla `MoodLog` + migración Prisma + refactor de `PUT /api/profile/me` con validación zod + transacción + endpoint `GET /api/profile/mood-history`. Frontend añade catálogos únicos (`moods.ts`, `avatarCatalog.ts`, `dailyPhrases.ts`), componente `AvatarPicker` reutilizable que reemplaza código duplicado en Settings/StepProfile, y nuevos componentes mood (`MoodSelectorSheet`, `MoodPairCard`, `MoodNudge`, `MyMoodWeek`). Cascada de frases por urgencia emocional con determinismo `cyrb53(coupleId-dayKey-category)`.

**Tech Stack:** TypeScript · React 18 · Vite · Prisma + SQLite/Postgres · Express · Zod · React Query · Vitest + RTL · Jest · Tailwind. Sin nuevas dependencias.

**Pre-flight requirement:** v1.5.1 (Supabase migrations reconcile) debe estar aplicado y `prisma migrate status` retornando "up to date" en producción antes de aplicar la migración nueva de v1.6.

**Spec source:** `docs/superpowers/specs/2026-04-26-v1.6-la-personalidad-design.md`

// src/frontend/src/components/v2/primitives/AvatarPicker.tsx
import { AVATAR_EMOJIS, AVATAR_COLORS } from '../../../data/avatarCatalog'

export interface AvatarPickerProps {
  emoji: string
  color: string
  onChange: (next: { emoji: string; color: string }) => void
  size?: 'sm' | 'md' | 'lg'
}

export function AvatarPicker({ emoji, color, onChange, size = 'md' }: AvatarPickerProps) {
  const previewSize = size === 'lg' ? 'h-24 w-24 text-5xl' : size === 'sm' ? 'h-12 w-12 text-2xl' : 'h-16 w-16 text-3xl'

  return (
    <div className="flex flex-col items-center gap-4">
      <div
        data-testid="avatar-preview"
        className={`${previewSize} rounded-full flex items-center justify-center shadow-lg`}
        style={{ background: `linear-gradient(135deg, ${color}, ${color}cc)` }}
      >
        {emoji}
      </div>

      <div className="grid grid-cols-6 gap-2 w-full">
        {AVATAR_EMOJIS.map(e => (
          <button
            key={e}
            data-testid={`emoji-opt-${e}`}
            data-selected={e === emoji}
            onClick={() => onChange({ emoji: e, color })}
            className={`text-2xl rounded-lg p-2 transition ${e === emoji ? 'bg-purple-500/30 ring-2 ring-purple-400' : 'hover:bg-white/5'}`}
            aria-label={`Avatar emoji ${e}`}
          >
            {e}
          </button>
        ))}
      </div>

      <div className="flex gap-2 overflow-x-auto w-full pb-1">
        {AVATAR_COLORS.map(c => (
          <button
            key={c.value}
            data-testid={`color-opt-${c.value}`}
            data-selected={c.value === color}
            onClick={() => onChange({ emoji, color: c.value })}
            className={`h-8 w-8 rounded-full flex-shrink-0 transition ${c.value === color ? 'ring-2 ring-white scale-110' : ''}`}
            style={{ background: c.value }}
            aria-label={`Color ${c.name}`}
          />
        ))}
      </div>
    </div>
  )
}
```

import { MOOD_KEYS } from '../data/moodKeys.js'

const profileUpdateSchema = z.object({
  surname: z.string().max(80).optional(),
  profilePhotoUrl: z.string().url().nullable().optional(),
  weeklyWorkHours: z.number().int().min(0).max(80).optional(),
  workMode: z.enum(['presencial','remoto','hibrido']).optional(),
  avatarEmoji: z.string().max(8).optional(),
  avatarColor: z.string().regex(/^#[0-9a-f]{6}$/i).optional(),
  currentMood: z.enum(MOOD_KEYS as unknown as [string, ...string[]]).nullable().optional(),
  hasCompletedOnboarding: z.boolean().optional(),
})

router.put('/me', async (req, res) => {
  const userId = (req as any).user.id
  const coupleId = (req as any).user.coupleId

  const parsed = profileUpdateSchema.safeParse(req.body)
  if (!parsed.success) {
    return res.status(400).json({ error: 'Datos inválidos', details: parsed.error.issues })
  }
  const body = parsed.data

  try {
    const result = await prisma.$transaction(async (tx) => {
      const existing = await tx.userProfile.findUnique({ where: { userId } })

      const profileFields: any = {}
      if (body.surname !== undefined) profileFields.surname = body.surname
      if (body.profilePhotoUrl !== undefined) profileFields.profilePhotoUrl = body.profilePhotoUrl
      if (body.weeklyWorkHours !== undefined) profileFields.weeklyWorkHours = body.weeklyWorkHours
      if (body.workMode !== undefined) profileFields.workMode = body.workMode

      if (body.currentMood !== undefined) {
        profileFields.currentMood = body.currentMood
        profileFields.moodUpdatedAt = body.currentMood ? new Date() : null
      }

      const profile = await tx.userProfile.upsert({
        where: { userId },
        update: profileFields,
        create: { userId, ...profileFields },
      })

      // User-level fields
      const userFields: any = {}
      if (body.avatarEmoji !== undefined) userFields.avatarEmoji = body.avatarEmoji
      if (body.avatarColor !== undefined) userFields.avatarColor = body.avatarColor
      if (body.hasCompletedOnboarding !== undefined) userFields.hasCompletedOnboarding = body.hasCompletedOnboarding
      if (Object.keys(userFields).length > 0) {
        await tx.user.update({ where: { id: userId }, data: userFields })
      }

      // MoodLog con anti-spam
      if (body.currentMood && body.currentMood !== existing?.currentMood) {
        const lastLog = await tx.moodLog.findFirst({
          where: { userId },
          orderBy: { createdAt: 'desc' },
        })
        const fiveMinAgo = Date.now() - 5 * 60 * 1000
        const shouldLog = !lastLog
          || lastLog.moodKey !== body.currentMood
          || lastLog.createdAt.getTime() < fiveMinAgo
        if (shouldLog) {
          await tx.moodLog.create({ data: { userId, coupleId, moodKey: body.currentMood } })
        }
      } else if (body.currentMood && body.currentMood === existing?.currentMood) {
        // Mismo mood: solo log si último log >5min
        const lastLog = await tx.moodLog.findFirst({
          where: { userId },
          orderBy: { createdAt: 'desc' },
        })
        const fiveMinAgo = Date.now() - 5 * 60 * 1000
        if (lastLog && lastLog.createdAt.getTime() < fiveMinAgo) {
          await tx.moodLog.create({ data: { userId, coupleId, moodKey: body.currentMood } })
        }
      }

      return profile
    })

    res.json(result)
  } catch (e: any) {
    console.error('PUT /profile/me failed', e)
    res.status(500).json({ error: 'Error al actualizar perfil' })
  }
})

  const { days, tz } = parsed.data

  try { new Intl.DateTimeFormat('en-US', { timeZone: tz }) } catch {

  }

  const since = new Date(Date.now() - (days + 1) * 24 * 60 * 60 * 1000)
  const logs = await prisma.moodLog.findMany({
    where: { userId, createdAt: { gte: since } },
    orderBy: { createdAt: 'asc' },
  })

  // agrupar por día local (TZ pasada), tomar el último
  const byDay = new Map<string, string>()
  const fmt = new Intl.DateTimeFormat('en-CA', { timeZone: tz, year: 'numeric', month: '2-digit', day: '2-digit' })
  for (const log of logs) {
    const dayKey = fmt.format(log.createdAt) // YYYY-MM-DD
    byDay.set(dayKey, log.moodKey) // sobrescribe con el último
  }

  // reconstruir array de N días terminando en hoy
  const history: Array<{ date: string; moodKey: string | null; emoji?: string; label?: string }> = []
  const today = new Date()
  for (let i = days - 1; i >= 0; i--) {
    const d = new Date(today.getTime() - i * 24 * 60 * 60 * 1000)
    const date = fmt.format(d)
    const moodKey = byDay.get(date) ?? null
    if (moodKey) {
      const m = MOOD_BY_KEY[moodKey]
      history.push({ date, moodKey, emoji: m?.emoji, label: m?.label })
    } else {
      history.push({ date, moodKey: null })
    }
  }

import { useMemo } from 'react'
import { MOOD_BY_KEY, type Mood } from '../data/moods'

export function useMoodVigent(
  moodKey: string | null | undefined,
  moodUpdatedAt: string | Date | null | undefined,
): Mood | null {
  return useMemo(() => {
    if (!moodKey || !moodUpdatedAt) return null
    const updated = typeof moodUpdatedAt === 'string' ? new Date(moodUpdatedAt) : moodUpdatedAt
    const ageMs = Date.now() - updated.getTime()
    if (ageMs >= 24 * 60 * 60 * 1000) return null
    return MOOD_BY_KEY[moodKey] ?? null
  }, [moodKey, moodUpdatedAt])
}

// src/frontend/src/utils/cyrb53.ts
export function cyrb53(str: string, seed = 0): number {
  let h1 = 0xdeadbeef ^ seed, h2 = 0x41c6ce57 ^ seed
  for (let i = 0, ch; i < str.length; i++) {
    ch = str.charCodeAt(i)
    h1 = Math.imul(h1 ^ ch, 2654435761)
    h2 = Math.imul(h2 ^ ch, 1597334677)
  }
  h1 = Math.imul(h1 ^ (h1 >>> 16), 2246822507)
  h1 ^= Math.imul(h2 ^ (h2 >>> 13), 3266489909)
  h2 = Math.imul(h2 ^ (h2 >>> 16), 2246822507)
  h2 ^= Math.imul(h1 ^ (h1 >>> 13), 3266489909)
  return 4294967296 * (2097151 & h2) + (h1 >>> 0)
}

// src/frontend/src/utils/dailyPhrase.ts
import { PHRASES, type Phrase, type PhraseCategory } from '../data/dailyPhrases'
import { cyrb53 } from './cyrb53'

export interface PhraseState {
  coupleId: string
  dayKey: string  // YYYY-MM-DD en TZ local
  hasOpenDisputeRecent: boolean
  streakBrokenLast24h: boolean
  recentMilestone: boolean
  weekendDay: boolean
  heavyWeekDetected: boolean
  partnerHighContribLastWeek: boolean
  isMonday: boolean
}

export function pickPhraseCategory(state: PhraseState): PhraseCategory {
  if (state.hasOpenDisputeRecent) return 'reconciliacion'
  if (state.streakBrokenLast24h) return 'animo'
  if (state.recentMilestone) return 'hito'
  if (state.weekendDay) return 'celebrar'
  if (state.heavyWeekDetected) return 'calma'
  if (state.partnerHighContribLastWeek) return 'agradecer'
  if (state.isMonday) return 'animo-suave'
  return 'neutra-positivo'
}

export function getDailyPhrase(state: PhraseState): Phrase {
  const category = pickPhraseCategory(state)
  const pool = PHRASES.filter(p => p.category === category)
  if (pool.length === 0) {
    // safety net (no debería pasar gracias al test del catálogo)
    return PHRASES[0]
  }
  const seed = cyrb53(`${state.coupleId}-${state.dayKey}-${category}`)
  return pool[seed % pool.length]
}
```

// src/frontend/src/hooks/useDailyPhraseState.ts
import { useMemo } from 'react'
import { useQuery } from '@tanstack/react-query'
import apiClient from '../services/apiClient'
import type { PhraseState } from '../utils/dailyPhrase'

interface Args { coupleId: string; tz?: string }

function todayKey(tz: string): string {
  return new Intl.DateTimeFormat('en-CA', { timeZone: tz, year: 'numeric', month: '2-digit', day: '2-digit' }).format(new Date())
}

export function useDailyPhraseState({ coupleId, tz = 'Europe/Madrid' }: Args): PhraseState {
  const eventsQ = useQuery({ queryKey: ['events', 'all'], queryFn: async () => (await apiClient.get('/events')).data })
  const tasksQ = useQuery({ queryKey: ['tasks', 'logs', 'recent'], queryFn: async () => (await apiClient.get('/tasks/logs?since=48h')).data })
  const achievementsQ = useQuery({ queryKey: ['achievements', 'recent'], queryFn: async () => (await apiClient.get('/achievements/user')).data })
  const streakQ = useQuery({ queryKey: ['streak'], queryFn: async () => (await apiClient.get('/gamification/status')).data })

  return useMemo<PhraseState>(() => {
    const now = new Date()
    const dow = new Intl.DateTimeFormat('en-US', { timeZone: tz, weekday: 'short' }).format(now)
    const isMonday = dow === 'Mon'
    const weekendDay = dow === 'Sat' || dow === 'Sun'

    const events = eventsQ.data ?? []
    const taskLogs = tasksQ.data ?? []
    const achievements = achievementsQ.data ?? []
    const streak = streakQ.data ?? { current: 0, previousDay: 0 }

    const hasOpenDisputeRecent =
      taskLogs.some((t: any) => t.status === 'disputed') ||
      events.some((e: any) => e.status === 'rejected' && new Date(e.updatedAt).getTime() > Date.now() - 48 * 3600 * 1000)

    const streakBrokenLast24h = (streak.current ?? 0) === 0 && (streak.previousDay ?? 0) > 0

    const recentMilestone = achievements.some((a: any) => new Date(a.unlockedAt).getTime() > Date.now() - 24 * 3600 * 1000)
      || (streak.leveledUpRecently === true)

    // semana cargada: ≥4 eventos largos (>3h) o suma >25h en 7d
    const lastWeek = events.filter((e: any) => new Date(e.dateStart).getTime() > Date.now() - 7 * 86400000)
    const longEvents = lastWeek.filter((e: any) => (new Date(e.dateEnd).getTime() - new Date(e.dateStart).getTime()) > 3 * 3600 * 1000)
    const sumHours = lastWeek.reduce((s: number, e: any) => s + (new Date(e.dateEnd).getTime() - new Date(e.dateStart).getTime()) / 3600000, 0)
    const heavyWeekDetected = longEvents.length >= 4 || sumHours > 25

    // partner alto aporte: simplificado — el partner > 60% de pointsCalculated en últimos 7 días
    const partnerPoints = lastWeek.filter((e: any) => e.createdBy !== /* myUserId */ '__never').reduce((s: number, e: any) => s + (e.pointsCalculated ?? 0), 0)
    const totalPoints = lastWeek.reduce((s: number, e: any) => s + (e.pointsCalculated ?? 0), 0)
    const partnerHighContribLastWeek = totalPoints > 0 && (partnerPoints / totalPoints) > 0.6

    return {
      coupleId,
      dayKey: todayKey(tz),
      hasOpenDisputeRecent,
      streakBrokenLast24h,
      recentMilestone,
      weekendDay,
      heavyWeekDetected,
      partnerHighContribLastWeek,
      isMonday,
    }
  }, [coupleId, tz, eventsQ.data, tasksQ.data, achievementsQ.data, streakQ.data])
}
