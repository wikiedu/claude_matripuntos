**Goal:** Convertir Matripuntos en experiencia con motivación intrínseca: niveles de pareja, achievements reales, streaks blandos, retos semanales, replays "hace 1 año", web push opt-in.

**Tech Stack:** Prisma + Postgres (additions), Express routes, React + Zustand state, web-push library, react-query mutations.

```typescript
// src/backend/src/services/levelService.ts
import { LEVEL_THRESHOLDS, LEVEL_NAMES, LEVEL_PERKS } from '../data/levelTable.js'

export function computeLevel(xp: number): { level: number; name: string; perks: string[]; threshold: number; nextThreshold: number } {
  let level = 1
  for (let i = LEVEL_THRESHOLDS.length - 1; i >= 0; i--) {
    if (xp >= LEVEL_THRESHOLDS[i]) {
      level = i + 1
      break
    }
  }
  level = Math.min(level, 10)
  const name = LEVEL_NAMES[level - 1]
  const perks = LEVEL_PERKS[level] ?? []
  const threshold = LEVEL_THRESHOLDS[level - 1]
  const nextThreshold = level < 10 ? LEVEL_THRESHOLDS[level] : threshold
  return { level, name, perks, threshold, nextThreshold }
}

export function xpToNext(xp: number): number {
  const { level, nextThreshold } = computeLevel(xp)
  if (level === 10) return 0
  return nextThreshold - xp
}
```
