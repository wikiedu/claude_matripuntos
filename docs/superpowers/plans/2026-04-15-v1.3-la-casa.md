```typescript
import prisma from '../lib/prisma.js'
import type { Task } from '@prisma/client'

type Frequency = 'daily' | 'biweekly' | 'weekly' | 'bimonthly' | 'monthly'

interface OccurrenceOptions {
  maxCount?: number
  endDate?: Date
}

/**
 * Pure function: given a start date and frequency, returns the next N occurrence dates.
 * Exported for testing. Does NOT touch the database.
 */
export function getNextOccurrences(
  start: Date,
  frequency: Frequency,
  weeks: number,
  options: OccurrenceOptions = {}
): Date[] {
  const results: Date[] = []
  const endDate = options.endDate ?? new Date(start.getTime() + weeks * 7 * 86400000)
  const maxCount = options.maxCount ?? Infinity
  let current = new Date(start)

  while (current <= endDate && results.length < maxCount) {
    results.push(new Date(current))
    switch (frequency) {
      case 'daily':
        current = new Date(current.getTime() + 86400000)
        break
      case 'biweekly':
        current = new Date(current.getTime() + 2 * 86400000)
        break
      case 'weekly':
        current = new Date(current.getTime() + 7 * 86400000)
        break
      case 'bimonthly':
        current = new Date(current.getTime() + 14 * 86400000)
        break
      case 'monthly': {
        const next = new Date(current)
        next.setMonth(next.getMonth() + 1)
        current = next
        break
      }
    }
  }

  return results
}

/**
 * Generate TaskLog instances (isAutoGenerated=true) for a recurring task
 * for the next 8 weeks. Skips dates that already have an auto-generated log.
 */
export async function generateOnCreate(task: Task): Promise<void> {
  if (!task.isRecurring || !task.frequency || !task.recurrenceStart) return

  const start = task.recurrenceStart
  const options: OccurrenceOptions = {}
  if (task.recurrenceEnd) options.endDate = task.recurrenceEnd
  if (task.maxOccurrences) options.maxCount = task.maxOccurrences - task.occurrenceCount

  const dates = getNextOccurrences(start, task.frequency as Frequency, 8, options)
  if (dates.length === 0) return

  // Fetch existing auto-generated logs to avoid duplicates
  const existingLogs = await prisma.taskLog.findMany({
    where: { taskId: task.id, isAutoGenerated: true },
    select: { scheduledFor: true },
  })
  const existingDates = new Set(existingLogs.map(l => l.scheduledFor?.toISOString()))

  const toCreate = dates.filter(d => !existingDates.has(d.toISOString()))
  if (toCreate.length === 0) return

  await prisma.taskLog.createMany({
    data: toCreate.map(d => ({
      coupleId: task.coupleId,
      taskId: task.id,
      date: d,
      scheduledFor: d,
      isAutoGenerated: true,
      pointsBase: task.pointsBase,
      pointsFinal: task.pointsBase,
      status: 'pending',
    })),
    skipDuplicates: true,
  })

  await prisma.task.update({
    where: { id: task.id },
    data: { occurrenceCount: { increment: toCreate.length } },
  })
}

/**
 * Extend the sliding window by one week for all recurring tasks of a couple.
 * Called by the Monday cron job.
 */
export async function generateInstancesForCouple(coupleId: string): Promise<void> {
  const tasks = await prisma.task.findMany({
    where: { coupleId, isRecurring: true, frequency: { not: null } },
  })
  for (const task of tasks) {
    await generateOnCreate(task)
  }
}

/**
 * Run weekly generation for ALL couples with active recurring tasks.
 * Called by cron every Monday at 08:00.
 */
export async function runWeeklyGeneration(): Promise<void> {
  const couples = await prisma.couple.findMany({
    where: { tasks: { some: { isRecurring: true } } },
    select: { id: true },
  })
  for (const couple of couples) {
    await generateInstancesForCouple(couple.id).catch(err =>
      console.error(`recurringTaskService error for couple ${couple.id}:`, err)
    )
  }
}
```

```typescript
import prisma from '../lib/prisma.js'
import { createNotification } from './notificationService.js'

export interface DigestData {
  weekStart: string
  weekEnd: string
  tasksCompleted: number
  user1Name: string
  user1Balance: number
  user2Name: string
  user2Balance: number
  achievementsUnlocked: string[]
  dailyStreakDays: number
}

function getLastWeekRange(): { start: Date; end: Date } {
  const now = new Date()
  const dayOfWeek = now.getDay() // 0=Sun, 1=Mon...
  const daysToLastMonday = dayOfWeek === 0 ? 6 : dayOfWeek - 1
  const lastMonday = new Date(now)
  lastMonday.setDate(now.getDate() - daysToLastMonday - 7)
  lastMonday.setHours(0, 0, 0, 0)
  const lastSunday = new Date(lastMonday)
  lastSunday.setDate(lastMonday.getDate() + 6)
  lastSunday.setHours(23, 59, 59, 999)
  return { start: lastMonday, end: lastSunday }
}

async function buildDigestForCouple(coupleId: string): Promise<DigestData | null> {
  const { start, end } = getLastWeekRange()

  const couple = await prisma.couple.findUnique({
    where: { id: coupleId },
    include: { users: true },
  })
  if (!couple || couple.users.length < 2) return null

  const [user1, user2] = couple.users

  // Tasks completed last week
  const tasksCompleted = await prisma.taskLog.count({
    where: {
      coupleId,
      status: { in: ['verified', 'pending'] },
      date: { gte: start, lte: end },
      isAutoGenerated: false,
    },
  })

  // Points balance for each user
  const user1Tx = await prisma.pointsTransaction.aggregate({
    where: { coupleId, userId: user1.id },
    _sum: { amount: true },
  })
  const user2Tx = await prisma.pointsTransaction.aggregate({
    where: { coupleId, userId: user2.id },
    _sum: { amount: true },
  })

  // Achievements unlocked last week
  const achievements = await prisma.coupleAchievement.findMany({
    where: {
      coupleId,
      unlockedAt: { gte: start, lte: end },
    },
    include: { definition: { select: { name: true } } },
  })

  return {
    weekStart: start.toISOString(),
    weekEnd: end.toISOString(),
    tasksCompleted,
    user1Name: user1.name,
    user1Balance: Number(user1Tx._sum.amount ?? 0),
    user2Name: user2.name,
    user2Balance: Number(user2Tx._sum.amount ?? 0),
    achievementsUnlocked: achievements.map(a => a.definition.name),
    dailyStreakDays: couple.dailyStreakDays,
  }
}

export async function sendWeeklyDigests(): Promise<void> {
  const couples = await prisma.couple.findMany({ select: { id: true } })

  for (const { id: coupleId } of couples) {
    try {
      const data = await buildDigestForCouple(coupleId)
      if (!data) continue

      const couple = await prisma.couple.findUnique({
        where: { id: coupleId },
        include: { users: true },
      })
      if (!couple) continue

      const message = JSON.stringify(data)
      const title = `📬 Tu resumen semanal — ${data.tasksCompleted} tareas, racha de ${data.dailyStreakDays} días`

      for (const user of couple.users) {
        await createNotification({
          coupleId,
          userId: user.id,
          type: 'weekly_digest',
          title,
          message,
        })
      }
    } catch (err) {
      console.error(`digestService error for couple ${coupleId}:`, err)
    }
  }
}
```

```typescript
import express, { Request, Response } from 'express'
import { authenticateToken } from '../middleware/auth.js'
import { z } from 'zod'
import prisma from '../lib/prisma.js'

const router = express.Router()

const addItemSchema = z.object({ text: z.string().min(1) })
const updateItemSchema = z.object({
  isChecked: z.boolean().optional(),
  text: z.string().min(1).optional(),
})

// GET /api/shopping — get active list + last 4 archived
router.get('/', authenticateToken, async (req: Request, res: Response): Promise<void> => {
  try {
    const coupleId = req.coupleId!

    // Ensure active list exists
    let active = await prisma.shoppingList.findFirst({
      where: { coupleId, isActive: true },
      include: { items: { orderBy: { createdAt: 'asc' } } },
    })
    if (!active) {
      active = await prisma.shoppingList.create({
        data: { coupleId },
        include: { items: true },
      })
    }

    const history = await prisma.shoppingList.findMany({
      where: { coupleId, isActive: false },
      include: { items: { orderBy: { createdAt: 'asc' } } },
      orderBy: { archivedAt: 'desc' },
      take: 4,
    })

    res.json({ active, history })
  } catch (err) {
    console.error('GET /shopping error:', err)
    res.status(500).json({ error: 'Error al cargar la lista de la compra' })
  }
})

// POST /api/shopping/items — add item to active list
router.post('/items', authenticateToken, async (req: Request, res: Response): Promise<void> => {
  try {
    const coupleId = req.coupleId!
    const { text } = addItemSchema.parse(req.body)

    let list = await prisma.shoppingList.findFirst({ where: { coupleId, isActive: true } })
    if (!list) {
      list = await prisma.shoppingList.create({ data: { coupleId } })
    }

    const item = await prisma.shoppingItem.create({
      data: { listId: list.id, text },
    })
    res.status(201).json(item)
  } catch (err) {
    if (err instanceof z.ZodError) { res.status(400).json({ error: err.errors[0].message }); return }
    console.error('POST /shopping/items error:', err)
    res.status(500).json({ error: 'Error al añadir item' })
  }
})

// PUT /api/shopping/items/:id — update item (check/uncheck or rename)
router.put('/items/:id', authenticateToken, async (req: Request, res: Response): Promise<void> => {
  try {
    const coupleId = req.coupleId!
    const updates = updateItemSchema.parse(req.body)

    const item = await prisma.shoppingItem.findFirst({
      where: { id: req.params.id, list: { coupleId } },
    })
    if (!item) { res.status(404).json({ error: 'Item no encontrado' }); return }

    const data: Record<string, unknown> = { ...updates }
    if (updates.isChecked !== undefined) {
      data.checkedBy = updates.isChecked ? req.userId : null
      data.checkedAt = updates.isChecked ? new Date() : null
    }

    const updated = await prisma.shoppingItem.update({ where: { id: item.id }, data })
    res.json(updated)
  } catch (err) {
    if (err instanceof z.ZodError) { res.status(400).json({ error: err.errors[0].message }); return }
    res.status(500).json({ error: 'Error al actualizar item' })
  }
})

// DELETE /api/shopping/items/:id
```

```tsx
import type { TaskSchedule } from '../types'

interface Props {
  value: TaskSchedule | null
  onChange: (schedule: TaskSchedule | null) => void
}

const FREQUENCY_OPTIONS = [
  { value: '', label: 'Sin repetición' },
  { value: 'daily', label: 'Cada día' },
  { value: 'biweekly', label: 'Cada 2 días' },
  { value: 'weekly', label: 'Cada semana' },
  { value: 'bimonthly', label: 'Cada 2 semanas' },
  { value: 'monthly', label: 'Cada mes' },
]

export function TaskScheduleForm({ value, onChange }: Props) {
  const [enabled, setEnabled] = useState(!!value)

  const toggle = () => {
    const next = !enabled
    setEnabled(next)
    if (!next) onChange(null)
  }

  const update = (patch: Partial<TaskSchedule>) => {
    if (!value) return
    onChange({ ...value, ...patch })
  }

  return (
    <div style={{ marginTop: 12 }}>
      {/* Toggle */}
      <button
        type="button"
        onClick={toggle}
        style={{
          display: 'flex', alignItems: 'center', gap: 8,
          background: 'none', border: 'none', cursor: 'pointer', padding: 0, marginBottom: enabled ? 12 : 0,
        }}
      >
        <div style={{
          width: 36, height: 20, borderRadius: 10, position: 'relative',
          background: enabled ? 'rgba(245,158,11,0.6)' : 'rgba(255,255,255,0.1)',
          transition: 'background 0.2s',
        }}>
          <div style={{
            position: 'absolute', top: 2, left: enabled ? 18 : 2,
            width: 16, height: 16, borderRadius: 8,
            background: '#fff', transition: 'left 0.2s',
          }} />
        </div>
        <span style={{ fontSize: 12, color: 'var(--matri-text-2)' }}>Planificar esta tarea</span>
      </button>

      {enabled && (
        <div style={{ display: 'flex', flexDirection: 'column', gap: 8 }}>
          {/* Date+time */}
          <input
            type="datetime-local"
            value={value?.scheduledFor ?? ''}
            onChange={e => {
              const v = e.target.value
              if (!value) onChange({ scheduledFor: v })
              else update({ scheduledFor: v })
            }}
            style={{
              background: 'rgba(255,255,255,0.05)',
              border: '1px solid var(--matri-card-border)',
              borderRadius: 8, padding: '8px 10px',

          <select
            value={value?.frequency ?? ''}

              const freq = e.target.value
              if (!value?.scheduledFor) return
              if (!freq) onChange({ scheduledFor: value.scheduledFor })
              else update({ frequency: freq as TaskSchedule['frequency'] })
            }}
            style={{
              background: 'rgba(255,255,255,0.05)',
              border: '1px solid var(--matri-card-border)',
              borderRadius: 8, padding: '8px 10px',
              color: 'var(--matri-text)', fontSize: 12,
            }}
          >
            {FREQUENCY_OPTIONS.map(o => (
              <option key={o.value} value={o.value}>{o.label}</option>
            ))}
          </select>

          {/* End condition (only if recurring) */}
          {value?.frequency && (
            <div style={{ display: 'flex', gap: 8 }}>
              <input
                type="date"
                value={value.recurrenceEnd ?? ''}
                onChange={e => update({ recurrenceEnd: e.target.value || undefined })}
                placeholder="Termina el..."
                style={{
                  flex: 1, background: 'rgba(255,255,255,0.05)',
                  border: '1px solid var(--matri-card-border)',

              />

                placeholder="Máx. veces"
```
